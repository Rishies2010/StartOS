Make a folder called crosstoolset in the StartOS directory. Then do this in the crosstoolset folder:

wget https://ftp.tsukuba.wide.ad.jp/software/gcc/releases/gcc-14.2.0/gcc-14.2.0.tar.gz
wget https://ftp.gnu.org/gnu/binutils/binutils-2.44.tar.gz
tar -xf binutils-2.44.tar.gz && tar -xf gcc-14.2.0.tar.gz
mkdir build-binutils
cd build-binutils
../binutils-2.44/configure --target=i686-elf --prefix=$HOME/cross-tools --with-sysroot --disable-nls --disable-werror
make -j$(nproc)
make install

Now go again into the crosstoolset folder and:

mkdir build-gcc
cd build-gcc
../gcc-14.2.0/configure --target=i686-elf --prefix=$HOME/cross-tools --disable-nls --enable-languages=c,c++ --without-headers
make all-gcc -j$(nproc)
make all-target-libgcc -j$(nproc)
make install-gcc
make install-target-libgcc
echo 'export PATH="$HOME/cross-tools/bin:$PATH"' >> ~/.bashrc
source ~/.bashrc
